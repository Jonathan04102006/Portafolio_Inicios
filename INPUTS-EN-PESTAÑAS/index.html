1 - Input de correo electrónico

✅ Ejemplo
🧱 HTML
🎨 CSS
⚙️ JavaScript
<input type="email" id="correo" placeholder="ejemplo@.com" oninput="validarCorreo()">
<div id="mensaje-correo"></div>
      
input[type="email"] {
  padding: 10px;
  border-radius: 5px;
  border: 2px solid #ccc;
  width: 300px;
  font-size: 16px;
}
input.error {
  border-color: red;
}
.mensaje {
  color: red;
  font-size: 14px;
  margin-top: 5px;
}
      
function validarCorreo() {
  const input = document.getElementById('correo');
  const mensaje = document.getElementById('mensaje-correo');
  if (input.value.trim() === '') {
    mensaje.textContent = 'Campo requerido.';
    input.classList.add('error');
  } else {
    mensaje.textContent = '';
    input.classList.remove('error');
  }
}

3 - Input de búsqueda y dirección de web

✅ Ejemplo
🧱 HTML
🎨 CSS
⚙️ JavaScript
<input type="search" id="search" name="search" 
       placeholder="Escribir búsqueda o dirección de web" 
       oninput="validarBusqueda()">
<div id="mensaje-busqueda"></div>
    
input[type="search"] {
  padding: 10px;
  border-radius: 5px;
  border: 2px solid #ccc;
  width: 300px;
  font-size: 16px;
}
input.error {
  border-color: red;
}
.mensaje {
  color: red;
  font-size: 14px;
  margin-top: 5px;
}
    
function validarBusqueda() {
  const input = document.getElementById('search');
  const mensaje = document.getElementById('mensaje-busqueda');
  const valor = input.value.trim();

  const regexURL = /^(https?:\/\/)?(www\.)?[\w\-]+\.[a-z]{2,}(\S*)?$/i;

  if (valor === "") {
    mensaje.textContent = "Campo requerido.";
    input.classList.add("error");
  } else if (!regexURL.test(valor) && valor.length < 3) {
    mensaje.textContent = "Ingrese una dirección web válida o una búsqueda más extensa.";
    input.classList.add("error");
  } else {
    mensaje.textContent = "";
    input.classList.remove("error");
  }
}

function validarTelefono() {
  const input = document.getElementById("tel");
  const mensaje = document.getElementById("mensaje-telefono");
  const valor = input.value.trim();

  const regex = /^(\+?\d{1,3}[- ]?)?\d{8,14}$/;

  if (valor === "") {
    mensaje.textContent = "Campo requerido.";
    input.classList.add("error");
  } else if (!regex.test(valor)) {
    mensaje.textContent = "Número de teléfono inválido. Ej: +503 71234567.";
    input.classList.add("error");
  } else {
    mensaje.textContent = "";
    input.classList.remove("error");
  }
}

function validarURL() {
  const input = document.getElementById('url');
  const mensaje = document.getElementById('mensaje-url');
  const valor = input.value.trim();

  const regex = /^(https?:\/\/)[^\s/$.?#].[^\s]*$/i;

  if (valor === '') {
    mensaje.textContent = 'Campo requerido.';
    input.classList.add('error');
  } else if (!regex.test(valor)) {

    input.classList.add('error');
  } else {
    mensaje.textContent = '';
    input.classList.remove('error');
  }
}

6 - Input de rango

✅ Ejemplo
🧱 HTML
🎨 CSS
⚙️ JavaScript

250000 
<label for="price">(Ejemplo) Elija un precio de venta:</label><br>
<input 
  type="range" 
  name="price" 
  id="price" 
  min="50000" 
  max="500000" 
  step="100" 
  value="250000"
  oninput="actualizarRango()"
/>
<output class="price-output" id="output-price">250000</output>
<div id="mensaje-rango"></div>
    
input[type="range"] {
  width: 300px;
  background: linear-gradient(to right, #4caf50 var(--valor), #ccc var(--valor));
  height: 8px;
  border-radius: 4px;
  appearance: none;
  outline: none;
}
input[type="range"]::-webkit-slider-thumb {
  appearance: none;
  width: 18px;
  height: 18px;
  background: #4caf50;
  border-radius: 50%;
  cursor: pointer;
}
input.error {
  outline: 2px solid red;
}
.price-output {
  font-weight: bold;
  margin-left: 10px;
}
.mensaje {
  color: red;
  font-size: 14px;
  margin-top: 5px;
}
    
const rango = document.getElementById("price");

function actualizarRango() {
  const salida = document.getElementById("output-price");
  const mensaje = document.getElementById("mensaje-rango");
  const valor = parseInt(rango.value);
  salida.textContent = `$${valor.toLocaleString()}`;

  const min = parseInt(rango.min);
  const max = parseInt(rango.max);
  const porcentaje = ((valor - min) / (max - min)) * 100;

  rango.style.setProperty('--valor', `${porcentaje}%`);
  if (valor < 100000 || valor > 400000) {
    mensaje.textContent = "El valor debe estar entre $100,000 y $400,000.";
    rango.classList.add("error");
  } else {
    mensaje.textContent = "";
    rango.classList.remove("error");
  }
}
